def smape(val,pred):
        return np.mean((np.abs(np.array(val) - np.array(pred)))/((np.abs(np.array(val)) + np.abs(np.array(pred)))/2)) * 100
    
def final_smape(rougher_val,rougher_pred,final_val,final_pred):
    return 0.25 * smape(rougher_val,rougher_pred) + 0.75 * smape(final_val,final_pred)
