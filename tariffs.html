    return row

def income_count(row):
    total_income = 0
    
    mb_used = row['mb_used']
    messages_amount = row['messages_amount']
    total_duration = row['total_duration']
    messages_included = row['messages_included']
    mb_per_month_included = row['mb_per_month_included']
    minutes_included = row['minutes_included']
    rub_monthly_fee = row['rub_monthly_fee']
    rub_per_gb = row['rub_per_gb']
    rub_per_message = row['rub_per_message']
    rub_per_minute = row['rub_per_minute']
    
    total_income += rub_monthly_fee
    
    if (mb_used - mb_per_month_included) > 0:
        total_income += math.ceil((mb_used - mb_per_month_included) / 1024) * rub_per_gb
    
    if (messages_amount - messages_included) > 0:
        total_income += (messages_amount - messages_included) * rub_per_message
    
    if (total_duration - minutes_included) > 0:
        total_income += (total_duration - minutes_included) * rub_per_minute
    
    row['income'] = total_income
    return row
